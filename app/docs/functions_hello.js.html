/**
 * HELLO
 * Returns a success message for testing successful connection through the API Gateway.
 * @module app/API/hello
 * @param {Object} event - Event received when the lambda function is invoked.
 * @returns {Object} The response object indicating success.
*/
import Logger from "../utils/custom-logger";

const fnName = `HELLO`;
const logger = new Logger(`API (${fnName})`);

export async function hello(event) {
    logger.req(fnName, event);

    const response = {
        statusCode: 418,
        body: JSON.stringify({
            message: `Short and stout!`,
        }),
    };

    logger.res(fnName, response);
    return response;
};
